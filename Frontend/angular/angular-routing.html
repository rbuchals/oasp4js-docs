Why do we need routers?

Simple state

configuration


      angular.module('myFirst', []).config(function ($stateProvider, $urlRouterProvider, $locationProvider) {
          $stateProvider.state('book-details', {
              url: '/book-mgmt/book',
              templateUrl: 'book-mgmt/book-details/book-details.html',
              controller: 'BookDetailsCtrl',
              controllerAs: 'bookDetailsCtrl'
          });

          $urlRouterProvider.when('', '/book-mgmt/book'); // set the book-details state as default
          $locationProvider.html5Mode(false); // set the hash bang mode
      });

URL Parameters / Resolve


    angular.module('myFirst', []).config(function ($stateProvider) {
        $stateProvider.state('book-details', {
            url: '/book-mgmt/book/bookId',
            templateUrl: 'book-mgmt/book-details/book-details.html',
            controller: 'BookDetailsCtrl',
            controllerAs: 'bookDetailsCtrl',
            resolve: {
                currentBook: function($http, $stateParams) {
                    return $http.get('services/rest/book/' + $stateParams.bookId).then(function (response) {
                        return response.data;
                    });
                }
            }
        });
    });

Resolved Objects / $state


    angular.module('myFirst', [])
        .controller('BookDetailsCtrl', function (currentBook, $state) {
            var bookDetailsCtrl = this;
            bookDetailsCtrl.currentBook = currentBook;

            bookDetailsCtrl.goToBookOverview = function () {
                $state.go('book-overview');
            }
    });
